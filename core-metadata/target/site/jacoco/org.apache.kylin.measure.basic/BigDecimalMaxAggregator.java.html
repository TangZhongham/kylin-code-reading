package org.apache.kylin.measure.basic;

import java.math.BigDecimal;

import org.apache.kylin.measure.MeasureAggregator;

/**
 */
@SuppressWarnings("serial")
public class BigDecimalMaxAggregator extends MeasureAggregator<BigDecimal> {

    BigDecimal max = null;

    @Override
    public void reset() {
        max = null;
    }

    @Override
    public void aggregate(BigDecimal value) {
        if (max == null)
            max = value;
        else if (max.compareTo(value) < 0)
            max = value;
    }

    @Override
    public BigDecimal aggregate(BigDecimal value1, BigDecimal value2) {
        if (value1 == null) {
            return value2;
        } else if (value2 == null) {
            return value1;
        }

        if (value1.compareTo(value2) > 0)
            return value1;
        else
            return value2;
    }

    @Override
    public BigDecimal getState() {
        return max;
    }

    @Override
    public int getMemBytesEstimate() {
        return guessBigDecimalMemBytes();
    }
}
